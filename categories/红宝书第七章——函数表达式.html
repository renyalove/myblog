# 红宝书第七章——函数表达式

# 1、定义函数

// 1，函数声明，变量提升
function Fun() { }
//2.函数表达式 ，不会变量提升
var Fun = function () { }

# 2、arguments.callee实现递归

function fun(n) { 
  if (n <= 1) return 1
  else return n*arguments.callee(n-1)
}

# 3、闭包——可访问另一个函数作用域中变量的函数

 function CreateFunctions() { 
  var res = new Array()
  for (var i = 0; i < 5; i++) { 
    res[i] = function (num) { 
      // i按值传递给num，每次都不一样
      return function () { 
        return num
      }
    }(i)
  }
  return res
}
console.log(CreateFunctions().forEach(function (item) { 
  console.log(item());// 0 1 2 3 4  
}));

# 4、关于this对象

  • this是在运行时基于函数的执行环境绑定的。

  • 匿名函数的this指向window，全局函数——this为window，函数作为对象方法——this为对象

//每个函数调用时都会取得两个变量——this和arguments，内部函数搜索变量时只会搜索到其活动变量（沿着作用域链找）
var name = "the window";
var object = {
  name: 'My object',
  getName: function () {
    // var that=this    
    return function () {  
     // return that.name // My object          
      return this.name //  the window
    }
  }
}

# 5、内存泄漏

一个避免内存泄漏的例子——避免循环引用

function assignHandle() { 
  var ele = document.getElementById("ele")
  // 保存ele 的id值
  var id = ele.id
  //! 这个闭包会引用包含函数的整个活动对象，其中有ele，虽然没直接引用
  ele.onclick = function () { 
    alert(id)
  }

  //!解除对ele 的引用
  ele=null
}

# 7、私有变量

在函数中定义的变量都是私有变量。

function MyObject() {
  // 私有函数和变量
  var privateVar = 10;
  function privateFn() {
    return false
  }
  // 特权方法，只能通过这个方法访问到私有函数和变量
  this.publicMethod = function () { 
    privateVar++
    return privateFn()
  }
}
console.log(MyObject.privateVar);//不能这么访问函数中的变量
// 实例化对象
var o = new MyObject()
console.log(o.privateVar);//undefined  访问不到
console.log(o.publicMethod());//false  可以访问到

（1）特权方法可以隐藏不该被直接修改的数据

function Person(name) { 
  // 构造函数的缺点——每个实例都会创建这组方法
   //只能通过这两个函数访问name属性 
  this.getName = function () { 
    return name;
  }
  this.setName = function (newName) { 
    name=newName
  }
}

（2）静态私有变量避免构造函数的问题——方法会多次创建。

但是name成为静态是为所有实例共享的属性，new Person 或 setName 都会改变这个值，会影响所有的实例。

(function () {
  var name = '';
  // 这里创建全局函数对象
  Person = function (value) {
    name = value    
  }
  Person.prototype.getName = function () {
    return name;
  }
  Person.prototype.setName = function (value) { 
    name=value
  }
})()

（3）模块化创建私有变量

用于实现单例——只有一个实例的对象

var singleton = function () { 
  // 私有变量和函数
  var privateVar = 10;
  function privateFn () { 
    return false
  }
  // 公有方法
  return {
    publicProperty:true,
    publicMethod: function () { 
      privateVar++;
      return privateFn()
    }
  }
}()

（4）增强模式的模块化创建私有变量，针对单例必须是某种类型的实例的情况

var singleton = function () { 
  // 私有变量和函数
  var privateVar = 10;
  function privateFn () { 
    return false
  }
  // 创建对象
  var  object=new Person('sera')
  // 公有方法
    object.publicProperty=true,
    object.publicMethod=function () { 
      privateVar++;
      return privateFn()
    }
  // 返回副本
  return object
}()
