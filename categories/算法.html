                                # 长度最小的子数组

                                https://leetcode-cn.com/problems/minimum-size-subarray-sum/solution/javade-jie-fa-ji-bai-liao-9985de-yong-hu-by-sdwwld/

                                首先最容易想到的是暴力求解，使用两个for循环，一个for循环固定一个数字比如m，另一个for循环从m的下一个元素开始累加，当和大于等于s的时候终止内层循环，顺便记录下最小长度;

                                /**
 * @param {number} s
 * @param {number[]} nums
 * @return {number}
 */
var minSubArrayLen = function(s, nums) {
//
     var min=Number.MAX_VALUE;
     for(var i=0;i<nums.length;i++){
        var sum=nums[i];
        if(sum>=s) return 1
        for(var j=i+1;j<nums.length;j++){
            sum+=nums[j]
            if(sum>=s) {
                min=Math.min(min,j-i+1)
                break
            }
        }
     }
     return min==Number.MAX_VALUE?0:min;   
    
};

                                我们把数组中的元素不停的入队，直到总和大于等于s为止，接着记录下队列中元素的个数，然后再不停的出队，直到队列中元素的和小于s为止（如果不小于s，也要记录下队列中元素的个数，这个个数其实就是不小于s的连续子数组长度，我们要记录最小的即可）。接着再把数组中的元素添加到队列中……重复上面的操作，直到数组中的元素全部使用完为止。

                                var minSubArrayLen = function(s, nums) {    
    var min=Number.MAX_VALUE;
    var sum=0;
    //不直接使用队列，使用两个指针
    var hi=0,lo=0;
    //当数组中的数都用完之后结束
    while(hi<nums.length){
        //往前累加
        sum+=nums[hi++]
        //大于给定的数之后出队
        while(sum>=s){
            //先记录此时的长度
            min=Math.min(min,hi-lo)
            //出队，最头开始出
            sum-=nums[lo++]
        }
    }
        
    return min==Number.MAX_VALUE?0:min  

};

                                • 取跟节点为目标节点，开始遍历
                                • 1.左孩子入栈 -> 直至左孩子为空的节点
                                • 2.节点出栈 -> 访问该节点
                                • 3.以右孩子为目标节点，再依次执行1、2、3
                                var inorderTraversal = function(root) {
    var res=[]
    var stack=[]
    var current=root
    while(current|| stack.length>0){
        while(current){
            stack.push(current)
            current=current.left
        }
        current=stack.pop()
        res.push(current.val)
        current=current.right
    }
    return  res
};

                                # 删除链表倒数第k个数

                                第一个指针从列表的开头向前移动 n+1n+1 步，而第二个指针将从列表的开头出发。现在，这两个指针被 nn 个结点分开。我们通过同时移动两个指针向前来保持这个恒定的间隔，直到第一个指针到达最后一个结点。此时第二个指针将指向从最后一个结点数起的第 nn 个结点。我们重新链接第二个指针所引用的结点的 next 指针指向该结点的下下个结点。

                                var removeNthFromEnd = function(head, n) {
    var dumy = new ListNode(0)
    dumy.next=head
    var first = dumy
    var second= dumy

        first = first.next
        second = second.next
    }
    second.next = second.next.next

};

        this.cache = {}
    }
    makeChange(amount) {
        let min = []
        if (!amount) return []
        if (this.cache[amount]) return this.cache[amount]

            var leftAmount = amount - this.changeType[i]
            let newMin

                    newMin = this.makeChange(leftAmount)
            }
            if (leftAmount >= 0 && (newMin.length < min.length - 1 || !min.length)) { 
                min=[this.changeType[i]].concat(newMin)
            }
            return this.cache[amount] = min
        }
    }
}

                                # 4.归并排序

                                归并排序算法是将两个（或两个以上）有序表合并成一个新的有序表，即把待排序序列分为若干个子序列，每个子序列是有序的。然后再把有序子序列合并为整体有序序列。

                                归并排序和快排的思路类似，都是递归分治，区别在于快排边分区边排序，而归并在分区完成后才会排序。

                                // 归并排序
function mergeSort(arr) { 
    // 划分到只有一个元素的时候停止
    if (arr.length <= 1) return arr 
    // 否则就一直拆分
    var mid = arr.length / 2 | 0
    var leftArr = arr.slice(0, mid)
    var rigthArr = arr.slice(mid)
    // 先拆分后合并
    return merge(mergeSort(leftArr),mergeSort(rigthArr))    
}
// 合并两个有序数组成为新数组的思路
function merge(left, right) {
    var res = []
    while (left.length && right.length) { 
        left[0]<right[0]?res.push(left.shift()):res.push(right.shift())
    }
    while (left.length) { 
        res.push(left.shift())
    }
    while (right.length) {
        res.push(right.shift())
    }
    return res
}

// 完成一次建堆，从最后的节点开始比较，最大的位于最前面
function maxHeap(arr,size) { 
    for (let i = size - 1; i >= 0; i--) { 
        // 每次都使当前节点最大
        heapify(arr,i,size)
    }
}

function heapify(arr, currentParentNode, size) { 
    if (currentParentNode < size) { 
        var left = currentParentNode * 2
        var right = left + 1
        var max = currentParentNode
        if (left < size) { 
            if (arr[left] > arr[max]) { 
                max = left
            }
        }
        if (right < size) {
            if (arr[right] > arr[max]) { 
                max = right
            }
        }
        if (max !== currentParentNode) { 
            [arr[currentParentNode], arr[max]] = [arr[max], arr[currentParentNode]]
            // 继续比较
            heapify(arr,max,size)
            
        }
    }
}

            fn(...args)
        })
        return this
    }
    once(type, callback) { 
        //只触发一次执行，因此要删除这个事件
        let onceFun = function(...args) { 
            callback(...args)
            this.remove(type,onceFun)
        }
        this.on(type,onceFun)
    }
    remove(type, callback) { 
        let callbacks = this.events[type];
        //一种类型的事件可能绑定了多个不同的回调函数，根据类型移除事件
        this.events[type] = callbacks && callbacks.filter(fn !== callback);
        return this;
    }
}

    for (let i = 3; i <= n; i++) { 
        dp[i]=dp[i-1]+dp[i-2]
    }
    return dp[n]
}
